-- AGGERGATE FUNCTIONS
SHOW DATABASES;
CREATE DATABASE STUDENTSS;
USE STUDENTSS;
-- CREATE TABLE
CREATE TABLE STUDENTSS (
    id INT,
    name VARCHAR(50),
    marks INT
);

-- INSERT VALUES
INSERT INTO STUDENTSS VALUES (1, 'Asha', 80);
INSERT INTO STUDENTSS VALUES (2, 'Ravi', 70);
INSERT INTO STUDENTSS VALUES (3, 'Neha', 90);
INSERT INTO STUDENTSS VALUES (4, 'Aman', 60);

SELECT * FROM STUDENTSS;

-- 1.COUNT()-  COUNT FUNCTIONN IS USED TO DETERMINE THE TOTAL NUMBER OF ROWS.
SELECT COUNT(*) FROM STUDENTSS;

-- 2.SUM()- CALCULATES THE TOTAL NUMBER OF NUMERIC VALUES IN A COLUMN.
SELECT SUM(marks) FROM STUDENTSS;

-- 3.AVG()- AVG FUNCTION IS USED TO FIND THE AVERAGE VALUE OF NUMERIC DATA IN A COLUMN.
SELECT AVG(marks) FROM STUDENTSS;

-- 4.MIN()- MIN FUNCTION IS USED TO IDENTIFY THE SMALLEST VALUE IN A DATA.
SELECT MIN(marks) FROM STUDENTSS;

-- 5.max()- MAX FUNCTION IS USED TO IDENTIFY THE HIGHEST VALUE IN A DATA.
SELECT MAX(marks) FROM STUDENTSS;
