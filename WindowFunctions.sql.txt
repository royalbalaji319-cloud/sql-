CREATE DATABASE EMPDB;
USE EMPDB;


CREATE TABLE EMPDB(
ID INT PRIMARY KEY,
NAME VARCHAR(50),
SALARY FLOAT
);

-- ROW_NUMBER()
SELECT
ID, 
    NAME,
    SALARY,
    ROW_NUMBER() OVER (ORDER BY SALARY DESC) AS row_num
FROM EMPDB;

SELECT 
NAME,
SALARY,
ROW_NUMBER() OVER (ORDER BY SALARY DESC) AS ROW_NUM FROM EMPDB;
-- RANK()
SELECT 
NAME,
SALARY,
RANK() OVER (ORDER BY SALARY DESC) AS ran FROM EMPDB;

-- DENSE_RANK()
SELECT 
NAME,
SALARY,
DENSE_RANK() OVER (ORDER BY SALARY DESC) FROM EMPDB;
